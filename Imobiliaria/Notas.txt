-- Geração de Modelo físico
-- Sql ANSI 2003 - brModelo.



CREATE TABLE proprietario (
id_proprietario int(11) PRIMARY KEY,
email varchar(255),
telefone varchar(255),
datanascimento date,
cpf varchar(255),
nome varchar(255)
banco varchar (255),
agencia varchar (255),
conta varchar (255),
pix varchar (255),
qtd_imovel int(11)
)

CREATE TABLE fiador (
nome varchar(255),
id_fiador int(11) PRIMARY KEY,
datanascimento date,
email varchar (255),
cpf varchar(255),
nome varchar(255)
)


CREATE TABLE aluguel (
preco decimal(6,2),
data date,
id_aluguel int(11) PRIMARY KEY,
id_corretor int(11),
id_imovel varchar(255),
id_inquilino int(11),
FOREIGN KEY(id_imovel) REFERENCES imovel (id_imovel)
)

CREATE TABLE imovel (
rua varchar(255),
bairro varchar(255),
cidade varchar(255),
estado varchar(255),
id_imovel varchar(255) PRIMARY KEY,
area int(11),
qtdquartos int(11),
qtdbanheiro int(11),
historico varchar(255),
garagem bit
)

CREATE TABLE inquilino (
datanascimento date,
nome varchar(255),
email varchar(255),
cpf varchar(255),
id_inquilino int(11) PRIMARY KEY,
id_fiador int(11)
FOREIGN KEY(id_fiador) REFERENCES fiador (id_fiador)
)

CREATE TABLE corretor (
id_corretor int(11) PRIMARY KEY,
datanascimento date,
nome varchar(255),
cpf varchar(255)
email varchar (255),
)

CREATE TABLE fotos (
id_foto int(11) PRIMARY KEY,
midia varchar(255),
id_imovel varchar(255),
FOREIGN KEY(id_imovel) REFERENCES imovel (id_imovel)
)

CREATE TABLE possui (
id_proprietario int(11),
id_imovel varchar(255),
FOREIGN KEY(id_imovel) REFERENCES imovel (id_imovel)
)


ALTER TABLE aluguel ADD FOREIGN KEY(id_corretor) REFERENCES corretor (id_corretor)
ALTER TABLE aluguel ADD FOREIGN KEY(id_inquilino) REFERENCES inquilino (id_inquilino)
ALTER TABLE inquilino ADD FOREIGN KEY(id_fiador) REFERENCES fiador (id_fiador)
ALTER TABLE possui ADD FOREIGN KEY(id_proprietario) REFERENCES proprietario (id_proprietario)
